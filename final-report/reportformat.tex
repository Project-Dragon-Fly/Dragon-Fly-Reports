%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
%%	PRINT MEMBER TABLE
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\newcommand{\membertable}
{	\renewcommand{\arraystretch}{1.25}
    \begin{table}[!ht]
        \centering
        \begin{tabular}{|c|l|}
            \hline
            \forloop{rr}{1}{\themembercnt>\value{rr}}{
            	\csname @memroll[\therr]\endcsname & 
            	\csname @memname[\therr]\endcsname \\ \hline
            }
    	    \csname @memroll[\therr]\endcsname & 
	        \csname @memname[\therr]\endcsname \\ \hline
         \end{tabular}
    \end{table}
	\renewcommand{\arraystretch}{1}
}

	}
	\csname @memname[\therr]\endcsname \space (\csname @memroll[\therr]\endcsname)
}
